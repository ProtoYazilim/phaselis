import React, { FC } from "react";
import { Pressable, Text } from "react-native";
import { ConsumerHOC } from "@protonative/provider";
import stylesheet from "./assets/styles";
import { LinkProps } from "./types";
import { router } from "expo-router";
import Slot from "../slot";
import useCombinedStyle from "@/hooks/useCombinedStyle";

const Link: FC<LinkProps> = ({
  contextValue,
  style,
  disabled,
  text,
  href,
  mode = "push",
  leftIcon,
  LeftSlot,
  rightIcon,
  RightSlot,
  primary,
  secondary,
  ...extraProps
}) => {
  const { getCombinedStyle } = useCombinedStyle(
    stylesheet,
    style,
    contextValue?.theme?.link,
    {
      disabled,
      primary,
      secondary,
      ...extraProps,
    },
  );

  return (
    <Pressable
      style={getCombinedStyle("container")}
      onPress={() => {
        router[mode](href);
      }}
    >
      <Slot style={getCombinedStyle("leftSlot")} icon={leftIcon} size="md">
        {LeftSlot && <LeftSlot />}
      </Slot>
      <Text style={getCombinedStyle("element")}>{text}</Text>
      <Slot style={getCombinedStyle("rightSlot")} icon={rightIcon} size="md">
        {RightSlot && <RightSlot />}
      </Slot>
    </Pressable>
  );
};

Link.displayName = "Link";

export default ConsumerHOC<LinkProps, LinkExtraProps>(Link);

import { LinkProps as ExpoLinkProps } from "expo-router/build/link/Link";
import { ViewStyle } from "react-native";

import { StyleProp, TextStyle } from "react-native";
import { SlotChildComponent, SlotIconName } from "../types";

interface LinkProps extends ExpoLinkProps {
  style?: StyleProp<TextStyle> & {
    container?: ViewStyle;
    element?: ViewStyle;
    leftSlot?: ViewStyle;
    rightSlot?: ViewStyle;
  };
  contextValue?: any;
  text?: string;
  mode?: "push" | "replace";
  leftIcon?: SlotIconName;
  rightIcon?: SlotIconName;
  LeftSlot?: SlotChildComponent;
  RightSlot?: SlotChildComponent;
  primary?: boolean;
  secondary?: boolean;
}

export { LinkProps };
