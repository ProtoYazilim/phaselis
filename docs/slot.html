import React from "react";
import LucideIcon from "../lucide-icon";
import { icons } from "lucide-react-native";

type SlotProps = {
  children?: any;
  icon?: keyof typeof icons;
  style?: any;
  size?: "xs" | "sm" | "md" | "lg" | "xl" | "xxl";
  strokeWidth?: number;
  onClick?: () => void;
  parentStates?: any;
};

const Slot: React.FC<SlotProps> = ({
  children,
  icon,
  style,
  size,
  strokeWidth,
  onClick,
  parentStates,
}) => {
  if (children) {
    const [defaultStyle, themeStyle, propStyle] = style;
    return (
      <>
        {React.cloneElement(children as React.ReactElement, {
          size: size || "md",
          defaultStyle,
          themeStyle,
          propStyle,
          parentStates,
          ...children?.props,
        })}
      </>
    );
  }
  if (icon) {
    return (
      <LucideIcon
        name={icon}
        style={style}
        size={size}
        strokeWidth={strokeWidth}
        onClick={onClick}
      />
    );
  }
  return null;
};

export default Slot;
