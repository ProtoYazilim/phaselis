import React from "react";
import { View, Text } from "react-native";
import { styles } from "./assets/styles";
import { IElementNativePropType } from "@protocommon/types";
import { ConsumerHOC } from "@protonative/provider";
import { EnumElementType } from "./type";

const Element = (props: IElementNativePropType) => {
  const styleObject = {
    abbr: {},
    address: styles.italic_style,
    b: styles.bold_style,
    blockquote: styles.blockquote_style,
    br: {},
    nav: {},
    canvas: {},
    caption: styles.caption_style,
    cite: styles.italic_style,
    dfn: styles.italic_style,
    em: styles.italic_style,
    embed: {},
    figcaption: {},
    h1: styles.h1_style,
    h2: styles.h2_style,
    h3: styles.h3_style,
    h4: styles.h4_style,
    h5: styles.h5_style,
    h6: styles.h6_style,
    hr: styles.hr_style,
    i: styles.italic_style,
    iframe: {},
    kbd: styles.kbd_style,
    legend: styles.legend_style,
    li: {},
    mark: styles.mark_style,
    object: {},
    p: {},
    q: {},
    source: {},
    span: {},
    strong: styles.bold_style,
    sub: {},
    sup: {},
    td: {},
    th: {},
    time: {},
  };

  const getDynamicStyle = (dynamicType: EnumElementType) => {
    return styleObject[dynamicType];
  };

  const {
    id,
    contextValue,
    className,
    type,
    onClick,
    children,
    style,
    ...otherProps
  } = props;
  const theme = contextValue?.theme?.element;
  switch (type) {
    case EnumElementType.NAV:
    case EnumElementType.IFRAME:
    case EnumElementType.CANVAS:
    case EnumElementType.HR:
      return (
        <View
          style={[
            getDynamicStyle(type as EnumElementType),
            theme,
            style?.element as React.CSSProperties,
          ]}
          {...otherProps}
        >
          {children}
        </View>
      );
    default:
      return (
        <Text
          style={[
            getDynamicStyle(type as EnumElementType),
            theme,
            style?.element as React.CSSProperties,
          ]}
          onPress={() => {
            if (onClick) {
              onClick();
            }
          }}
          {...otherProps}
        >
          {children}
        </Text>
      );
  }
};

Element.displayName = "Element";

Element.defaultProps = {
  type: EnumElementType.SPAN,
};

export default ConsumerHOC(Element);

import React from "react";
import { View, Modal as ModalNative, ViewStyle } from "react-native";
import Header from "./lib/Header";
import Body from "./lib/Body";
import Footer from "./lib/Footer";
import { IModalPropType } from "@protocommon/types";
import { styles } from "./assets/styles";

class Modal extends React.Component<IModalPropType> {
  static Header = Header;
  static Body = Body;
  static Footer = Footer;

  static displayName = "modal";
  static defaultProps = {};

  constructor(props: IModalPropType) {
    super(props);
  }

  render() {
    const { show, children, style, ...otherProps } = this.props;
    return (
      <ModalNative
        animationType="slide"
        transparent={true}
        disabled
        visible={show}
        {...otherProps}
      >
        <View style={[styles.proto_modal_overlay, style?.overlay as ViewStyle]}>
          <View
            style={[
              styles.proto_modal_container,
              style?.container as ViewStyle,
            ]}
          >
            {children}
          </View>
        </View>
      </ModalNative>
    );
  }
}

export default Modal;

import React from "react";
import { View } from "react-native";
import Loader from "./lib/Loader";
import { styles } from "./assets/styles";
import { IPanelPropTypes } from "@protocommon/types";
import Header from "./lib/Header";
import Body from "./lib/Body";
import Footer, { LeftFooter, RightFooter } from "./lib/Footer";
import Divider from "./lib/Divider";
import { ConsumerHOC } from "@protonative/provider";

const defaultProps: IPanelPropTypes = {
  shadow: false,
};

const renderLoader = (loading?: boolean, style?: any) => {
  return loading ? <Loader className={style && style.loading} /> : null;
};

const createPanelStyle = (shadow?: boolean, props?: any) => {
  const { style, contextValue } = props;
  const theme = contextValue.theme?.panel;

  return [
    styles.proto_panel,
    theme?.container,
    style && style.container,
    shadow && styles.proto_shadow,
  ];
};

const Panel = (props: IPanelPropTypes) => {
  const { loading, children, shadow } = props;

  return (
    <View style={createPanelStyle(shadow, props)}>
      {children}
      {renderLoader(loading, props.className)}
    </View>
  );
};

Panel.displayName = "Panel";
Panel.defaultProps = defaultProps;

Panel.Header = Header;
Panel.Body = Body;
Panel.Footer = Footer;
Panel.LeftFooter = LeftFooter;
Panel.RightFooter = RightFooter;
Panel.Divider = Divider;

export default ConsumerHOC(Panel);

import * as React from "react";
import { Text, View, ViewStyle } from "react-native";
import { IPanelBodyPropTypes } from "@protocommon/types";

import { ConsumerHOC } from "@protonative/provider";

const Body = (props: IPanelBodyPropTypes) => {
  const { style, noPadding, children, contextValue } = props;
  const theme = contextValue.theme?.panel;
  return (
    <View
      style={[
        styles.body,
        theme?.body,
        noPadding && styles.no_padding,
        style as ViewStyle,
      ]}
    >
      {React.Children.toArray(children).map((child, index) => {
        if (typeof child === "string") {
          return <Text key={index}>{child}</Text>;
        } else {
          return child;
        }
      })}
    </View>
  );
};

import * as React from "react";
import { Text, View, ViewStyle } from "react-native";
import { IPanelHeaderPropTypes } from "@protocommon/types";
import { styles } from "../assets/styles";
import { ConsumerHOC } from "@protonative/provider";

function createParentViewStyle({ style, contextValue }: any) {
  const theme = contextValue.theme?.panel;
  return [
    styles.proto_panel_header,
    theme?.header,
    style && style.container ? style.container : null,
  ];
}

function createTitleStyle() {
  return [styles.title_style];
}

function createTextStyle(className: any) {
  return [
    styles.proto_panel_caption,
    className && className.textStyle ? className.textStyle : null,
  ];
}

const Header = (props: IPanelHeaderPropTypes) => {
  const { title, children, style } = props;

  return (
    <View style={createParentViewStyle(props)}>
      <View style={createTitleStyle() as ViewStyle}>
        <Text style={createTextStyle(style)}>{title}</Text>
        <View>{children}</View>
      </View>
    </View>
  );
};
